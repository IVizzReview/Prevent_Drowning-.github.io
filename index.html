Prevent Drowning


บทคัดย่อ (Abstract)

เนื่องจากในปัจจุบันปัญหาเด็กจมน้ำนั้นมีอัตราที่เพิ่มขึ้นทุกปี และเด็กไทยเสียชีวิตจากการจมน้ำสูงเป็นอันดับหนึ่งเมื่อเทียบกับอุบัติเหตุอื่น ๆ ซึ่งสาเหตุหลักของอุบัติเหตุนี้คือ ผู้ปกครองของเด็กไม่ได้ดูแลเด็กอย่างใกล้ชิด ทั้งอุปกรณ์ช่วยป้องกันการจมน้ำในปัจจุบันมีขนาดใหญ่และไม่เหมาะสำหรับใส่พกพาไปยังสถานที่อื่น ทำให้เมื่อเกิดอุบัติเหตุจมน้ำ ผู้ปกครองไม่สามารถเข้าช่วยเหลือเด็กได้ทันท่วงที ส่งผลให้เด็กมีโอกาสรอดน้อยลง และเสียชีวิตในที่สุด ผู้พัฒนาจึงได้คิดค้นอุปกรณ์ช่วยป้องกันและเพื่อเพิ่มโอกาสรอดของเด็กจากการจมน้ำ โดยอุปกรณ์นี้จะช่วยแจ้งเตือนผู้ปกครองของเด็ก ทันทีที่เมื่ออุปกรณ์ตรวจวัดแรงดันน้ำรอบตัวของเด็กอยู่ในระดับที่สูงผิดปกติ อุปกรณ์จะทำการส่งข้อความแจ้งเตือนไปยังโทรศัพท์ของผู้ปกครอง ทำให้ผู้ปกครองรู้ตัวและสามารถเข้าช่วยเหลือเด็กได้ทันท่วงที

อุปกรณ์ (Equipment)

Arduino Nano 3.0 Mini USB

Arduino Nano 3.0 เหมือน Arduino Nano 3.0 รุ่นเดิมทุกอย่าง แต่ราคาถูกกว่า ต่างกันที่ใช้ชิฟ CH340G เป็นชิฟ USB Driver แทนรุ่น Arduino Nano 3.0 เดิม สามารถใช้ได้กับ Windows XP, Windows 7, Windows 8, Windows 10 ทั้ง 32/64bit, Linux, Mac OS ได้ และมีราคาถูกลง

SIM900A

SIM900A GSM/GPRS Module โมดูล SIM900 เป็นโมดูลซิมการ์ดที่สามารถใส่ซิมการ์ด มือถือ ควบคุมการ โทรเข้า โทรออก รับสาย วางสาย ส่งsms ได้ ใช้ไฟเลี้ยง 5V 2A
ขนาด: 5×5cm
อุณหภูมิในการทำงาน: -40°C ~ 85°C
น้ำหนัก: 3.4g
ช่วงแรงดันไฟฟ้า: 5V
การใช้พลังงานต่ำ: 1.5mA (sleep mode)

Pressure Transducer Sensor

เซ็นเซอร์ใช้ไฟเลี้ยง 5V ให้สัญญาณออกมาเป็น analog ใช่้งานง่ายเชื่อมต่อ arduino ได้โดยตรง ทำจากเหล็กแข็งแรง ตัวเซ็นเซอร์กันน้ำ
สามารถวัดความดันได้: 0-1.6MPa
ผู้ผลิต: DFRobot
อุณหภูมิในการทำงาน: -20°C ~ 85°C
ความดันการดำเนินงาน: 232.06PSI (1600 kPa)
ความดันสูงสุด: 290.08PSI (2000 kPa)









หลักการทำงาน (Processes)

เราจะใช้เซ็นเซอร์วัดแรงดัน Pressure Transducer Sensor วัดแรงดันว่าความลึกเท่านี้มีแรงดันน้ำประมาณเท่าไหร่ ถ้ามากกว่าที่กำหนดไว้ก็จะทำการส่งข้อความผ่าน SIM900A ไปหาผู้ปกครองเเละนับเวลาอีก 10 วินาที ถ้าหากเเรงดันยังมากกว่าที่กำหนดจะส่งข้อความใหม่อีกรอบ

Arduino Nano 3.0 Mini USB


#include <LiquidCrystal_I2C.h>
#include <SoftwareSerial.h>
LiquidCrystal_I2C lcd(0x27,16,2);
SoftwareSerial SIM900A(10,11);// TX|RX
int hour=12,minute=0,second=0;
int check=0,recall=0;
const float OffSet=0.483;
float V,P;
void setup() {//Just a setting
  lcd.begin();
  Serial.begin(9600);// Setting the baud rate of Serial Monitor (Arduino)
  SIM900A.begin(9600);// Setting the baud rate of GSM Module
  lcd.backlight();
  Serial.println("SIM900A Ready");
}
void loop() {
  V=analogRead(0)*5.00/1024;// Receive values from pressure sensors
  P=(V-OffSet)*400;// Convert to kilopascal
  second++;
  if (second>59){// Just a timer
    minute++;
    second=0;
  }
  if (minute>59){
    hour++;
    minute=0;
  }
  if (hour>23){
    hour=0;
  }
  lcd.clear();// Display to LCD screen
  lcd.print("TIME: ");
  lcd.print(hour);
  lcd.print(":");
  lcd.print(minute);
  lcd.print(":");
  lcd.print(second);
  lcd.setCursor(1,1);
  lcd.print("WHITEWHALEWATCH");
  Serial.print(P,1);
  Serial.println(" KPa");
  delay(990);
  if (P>20.0){// Sends a message when the P over 20 kilopascal
    SendMessage();
    second++;
    check=1;
  }
  while (check!=0){// Check if the child is still drowned
    recall++;
    second++;
    Serial.println(recall);
    delay(990);
    if (recall==10){
      check=0;
      recall=0;
    }
  }
}

void SendMessage(){
  // Fuction used to send messages
  Serial.println("Sending Message");
  SIM900A.println("AT+CMGF=1");// Sets the GSM Module in Text Mode
  delay(200);
  Serial.println("Set SMS Number");
  SIM900A.println("AT+CMGS=\"+660917499494\"\r");// Mobile phone number to send message
  delay(200);
  Serial.println("Set SMS Content");
  SIM900A.println("WHITEWHALEWATCH");// Messsage content
  delay(200);
  Serial.println("Finish");
  SIM900A.println((char)26);// ASCII code of CTRL+Z
  delay(400);
  Serial.println("Message has been sent");
}

#include <SoftwareSerial.h>
SoftwareSerial SIM900A(10,11);
void setup()
{
  SIM900A.begin(9600);// Setting the baud rate of GSM Module
  Serial.begin(9600);// Setting the baud rate of Serial Monitor (Arduino)
  Serial.println ("SIM900A Ready");
  delay(100);
  Serial.println ("Type s to send message or r to receive message");
}//Just setup
void loop()
{
  if (Serial.available()>0)
   switch(Serial.read())// Input text
  {
    case 's':// To send message
      SendMessage();
      break;
    case 'r':// To read message
      RecieveMessage();
      break;
  }
 if (SIM900A.available()>0)
   Serial.write(SIM900A.read());// Display result to Serial Monitor
}


 void SendMessage()
{
  // Fuction used to send messages
  Serial.println ("Sending Message");
  SIM900A.println("AT+CMGF=1");// Sets the GSM Module in Text Mode
  delay(1000);
  Serial.println ("Set SMS Number");
  SIM900A.println("AT+CMGS=\"+660917499494\"\r");// Mobile phone number to send message
  delay(1000);
  Serial.println ("Set SMS Content");
  SIM900A.println("Test");// Messsage content
  delay(100);
  Serial.println ("Finish");
  SIM900A.println((char)26);// ASCII code of CTRL+Z
  delay(1000);
  Serial.println ("Message has been sent");
}


 void RecieveMessage()
{
  // Fuction used to recieve messages
  Serial.println ("SIM900A Membaca SMS");
  delay (1000);
  SIM900A.println("AT+CNMI=2,2,0,0,0");// AT Command to receive a live SMS
  delay(1000);
  Serial.write ("Unread Message done");
}

const float OffSet = 0.483;
float V, P;
void setup() {
  Serial.begin(9600);

  Serial.print(V, 3);
  Serial.println("V");// Display V value to Serial Monitor

  Serial.print(P, 1);
  Serial.println("KPa");// Display P value to Serial Monitor

  delay(500);
}
